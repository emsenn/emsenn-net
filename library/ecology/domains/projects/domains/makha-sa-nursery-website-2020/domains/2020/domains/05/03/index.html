;; emsenn's land management elisp
;; (elme)
(require 'ht)

;; data
(setq elme-catalog (ht-create))

;; functions
(defun elme-add-specimen! (id info)
  (ht-set! elme-catalog id info))
(defun elme-get-specimen (id)
  (ht-get elme-catalog id))
(defun elme-get-specimen-attribute (id attr)
  (ht-get (ht-get elme-catalog id) attr))
(defun elme-get-specimen-species (id)
  (elme-get-specimen-attribute id 'species))

;; examples
(elme-add-specimen! 1 (ht ('species "foobar")))
(elme-get-specimen 1)
(elme-get-specimen-attribute 1 'species)
(elme-get-specimen-species 1)
